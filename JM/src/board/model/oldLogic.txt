/*
			 * 현재 페이지 = pageNum 
			 * 시작행 = StartRow 
			 * 종료행 = endRow 
			 * 총 페이지 수 = totalPageCount
			 */
			int pageNum = 0, startRow = 0, endRow = 0, totaPageCount = 0;
			
			// 만약 게시물의 숫자가 1개 이상이라도 존재한다면.
			if (articleTotalCount > 0) {

				if (nowPage != null) {
					// nowPage 가 null이면 페이징을 요청 한 곳에서 값을 가져왔음으로.
					startRow = (Integer.parseInt(nowPage) - 1) * VIEW_PAGE_PER_COUNT + 1;
				} else {
					// nowPage 가 null이면 페이징을 요청 한 곳에서 값을 가져오지 못했음으로.
					pageNum = 1;
					startRow = (pageNum - 1) * VIEW_PAGE_PER_COUNT + 1;
				}
				endRow = startRow + VIEW_PAGE_PER_COUNT - 1;
				// 총 페이지 수 = 게시물 총 숫자 / 페이지당 보여줄 숫자
				totaPageCount = articleTotalCount / VIEW_PAGE_PER_COUNT;
				// 만약 게시물 총 개수를 페이지당 보여 줄 숫자로 나눈 뒤
				// 나머지가 있으면 1페이지가 더 있는것임으로
				if (articleTotalCount % VIEW_PAGE_PER_COUNT > 0) {
					// 총페이지 수에 +1을 해준다.
					totaPageCount++;
				}
				// DAO에 
				dao.selectList(conn, startRow, VIEW_PAGE_PER_COUNT);
				
				
				// 게시물이 한개도 존재하지 않는다면.
			} else {
				pageNum = 0;
				
			}
